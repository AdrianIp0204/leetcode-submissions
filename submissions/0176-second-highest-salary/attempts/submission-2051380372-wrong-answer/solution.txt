import pandas as pd

def second_highest_salary(employee: pd.DataFrame) -> pd.DataFrame:
    employee.drop_duplicates(subset=['salary'], inplace=True)
    return pd.DataFrame({'SecondHighestSalary': [None]}) if len(employee) <= 1 else pd.DataFrame({'SecondHighestSalary': [employee.sort_values(by='salary').iloc[1, 1]]})
